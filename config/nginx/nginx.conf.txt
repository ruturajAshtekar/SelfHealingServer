events { }

http {
    upstream app_backend {
        server app:5000;
    }

    server {
        listen 80;

        # Main app traffic
        location / {
            proxy_pass http://app_backend;
        }

        # Health endpoint via load balancer
        location /health {
            proxy_pass http://app_backend/health;
        }

        # Prometheus metrics via load balancer
        location /metrics {
            proxy_pass http://app_backend/metrics;
        }
    }
}
