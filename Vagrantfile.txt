Vagrant.configure("2") do |config|
  # OS 
  config.vm.box = "ubuntu/jammy64"
  config.vm.hostname = "self-healing-server"

  # Port Forwarding
  config.vm.network "forwarded_port", guest: 5000, host: 5000, auto_correct: true  # Flask 
  config.vm.network "forwarded_port", guest: 8080, host: 8080, auto_correct: true  # Nginx
  config.vm.network "forwarded_port", guest: 9090, host: 9090, auto_correct: true  # Prometheus
  config.vm.network "forwarded_port", guest: 3000, host: 3000, auto_correct: true  # Grafana
  config.vm.network "forwarded_port", guest: 8081, host: 8081, auto_correct: true  # cAdvisor

  # VM Resources
  config.vm.boot_timeout = 600
  config.vm.provider "virtualbox" do |vb|
    vb.memory = "4096"
    vb.cpus   = 2
  end
  
  # shell script 
  config.vm.provision "shell", path: "provision.sh"
end
